def self.hash_for_table(state, term)
  cities_list = {}
  cities = City.where("code LIKE '#{state}%'").where("name LIKE '%#{term}%'").order(:code)
  cities.each do |city|
    cities_list[city.code] = {} unless cities_list[city.code]
    cities_list[city.code] = city.name
  end
  return cities_list
end
